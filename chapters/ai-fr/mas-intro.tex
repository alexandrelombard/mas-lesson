\begin{frame}[t]{{Systèmes multiagents :} Concepts clés}
	\begin{columns}
		\begin{column}{.8\linewidth}
			\begin{block}{Qu'est-ce qu'un agent ? \cite{Wooldridge09}}
				Une \Emph{entité autonome humaine, matérielle ou logicielle} qui :
				\begin{compactdescription}
				\item[Perçoit] son environnement (capteurs, messages, flux de données)
				\item[Raisonne] à l'aide de connaissances internes et d'objectifs
				\item[Agit] pour poursuivre ses objectifs (actionneurs, appels de services, décisions)
				\item[Communique] et \Emph{coopère} avec d'autres agents
				\end{compactdescription}
			\end{block}
			\begin{block}{Qu'est-ce qu'un système multiagents ? \cite{Wooldridge09}}
				\begin{compactdescription}
				\item[Décentralisé] aucun point de contrôle ou de défaillance unique
				\item[Évolutif] ajout/suppression d'agents sans reconcevoir l'ensemble du système
				\item[Hétérogène] les agents peuvent avoir des rôles, compétences et connaissances différents
				\item[Comportement émergent] résultats complexes issus de règles locales simples
				\end{compactdescription}
			\end{block}
		\end{column}
